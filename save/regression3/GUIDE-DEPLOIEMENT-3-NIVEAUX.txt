🔧 ÉTAPE 4: INTÉGRER AVEC payment-distribution-system.js
═══════════════════════════════════════════════════════════════════════════════

Dans payment-distribution-system.js, ajoute ceci:

const livreurLevels = require('./livreur-levels-release');

async function distributePaymentWithLevelAwareness(paymentIntentId, amount, deliveryId, livreurId) {
  try {
    // Déterminer le niveau du livreur
    const livreurLevel = await livreurLevels.getLivreurLevel(livreurId);
    
    console.log(`💰 Livreur Level: ${livreurLevel.name} - Déblocage: ${livreurLevel.paymentDelayHours}h`);

    // Distribuer le paiement normalement
    await distributePayment(paymentIntentId, amount, deliveryId);

    // Marquer avec le niveau
    const { error: levelError } = await supabase
      .from('payment_transactions')
      .update({
        livreur_level: livreurLevel.level,
        expected_release_at: new Date(Date.now() + livreurLevel.paymentDelayHours * 3600000)
      })
      .eq('delivery_id', deliveryId);

    return {
      success: true,
      level: livreurLevel,
      expectedReleaseIn: `${livreurLevel.paymentDelayHours}h`
    };

  } catch (error) {
    console.error('❌ Erreur:', error);
    return { success: false, error: error.message };
  }
}

📊 MONITORING
═══════════════════════════════════════════════════════════════════════════════

Query pour voir les déblocages:

SELECT 
  user_id,
  livreur_level,
  COUNT(*) as total_released,
  SUM(amount) as total_amount
FROM wallet
WHERE status = 'released'
GROUP BY user_id, livreur_level
ORDER BY total_amount DESC;

Query pour voir les paiements en attente:

SELECT 
  delivery_id,
  user_id,
  livreur_level,
  created_at,
  NOW() - created_at as elapsed
FROM wallet
WHERE status = 'pending'
ORDER BY created_at ASC;
